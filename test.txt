1. Создать index.html с базовой разметкой документа. Создать файл скрипта JS.
2. Подключить файл скрипта JS к документу.
3. В документе создать следующие элементы:
3.1. Поле для ввода имени placeholder="Имя"
3.2. Поле для ввода возраста
3.3. Кнопка Сохранить(<input type='button' value='Send'/>)
3.4. Кнопка Показать
3.5. Блок для вывода Результата
4. Создать следующую логику
4.1. Написать обработку данных из полей. Поле имя должно сохранить свое содержимое в this.name, поле возраст в this.age. Событие должно вызываться при нажатии на кнопку Сохранить
4.2. Написать вывод сохраненных данных в блок Результата по нажатию на кнопку Показать. Вывод должен быть формата:
Имя: <ИМЯ>, Возраст: <ВОЗРАСТ>

// 1. 
var name, age;
document.body.querySelector('[name=save]').onclick = function() {
    name = document.body.querySelector('[name=name]').value;
    age = document.body.querySelector('[name=age]').value;
}
document.body.querySelector('[name=load]').onclick = function() {
    document.getElementById('result').value = 'Имя: ' + name +', Возраст: ' + age;
}

//2.
function Personal() {
    return {
        Init: function() {
            this.age;
            this.name;
            this.body = document.body;
            this.obName = body.querySelectorAll('.class');
            this.obAge = body.querySelector('[name=age]');
            this.btnSave = body.querySelector('[name=save]');
            this.btnLoad = body.querySelector('[name=load]');
            this.obResult = body.querySelector('#result');

            this.BindEvents();

        },

        BindEvents: function() {
            this.btnSave.addEventListener('click', () => {
                this.age = this.obAge.value;
                this.name = this.obName.value;
            });

            this.btnLoad.addEventListener('click', () => {
                this.obResult.value = 'Имя: ' + this.name +', Возраст: ' + this.age;
            })
        }
    }
}
